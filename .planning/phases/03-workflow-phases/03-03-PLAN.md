```typescript
async getChangedFiles(fromHash: string): Promise<string[]> {
  try {
    const result = await this.git.raw(['diff', '--name-only', fromHash + '..HEAD']);
    return result.trim().split('\n').filter(Boolean);
  } catch {
    return [];
  }
}
```
