    2. After the featureContext gathering block (lines 71-86), add research summary gathering following the same try/catch pattern:
       ```typescript
       let researchSummaries: string | null = null;
       try {
         const researchDir = join(repoRoot, BRANCHOS_DIR, SHARED_DIR, RESEARCH_DIR);
         const artifacts = await readAllResearch(researchDir);
         if (artifacts.length > 0) {
           // Only include complete artifacts (per research recommendation)
           const complete = artifacts.filter(a => a.status === 'complete');
           // If workstream linked to feature, filter by feature relevance
           // Include artifacts with no feature link (general) plus feature-matched ones
           let relevant = complete;
           try {
             const metaPath = join(workstream.path, 'meta.json');
             const meta = await readMeta(metaPath);
             if (meta.featureId) {
               relevant = complete.filter(a => a.features.length === 0 || a.features.includes(meta.featureId!));
             }
           } catch {
             // meta already read above, reuse if possible; otherwise proceed with all complete
           }

           const summaryParts: string[] = [];
           for (const artifact of relevant) {
             const summary = extractSummary(artifact.body);
             if (summary) {
               summaryParts.push(`### ${artifact.topic} (${artifact.id})\n\n${summary}`);
             }
           }
           if (summaryParts.length > 0) {
             researchSummaries = summaryParts.join('\n\n');
           }
         }
       } catch {
         // Research dir missing or unreadable - proceed without research context
       }
       ```

       NOTE: The `meta` variable is already read in the featureContext block above. To avoid reading it twice, refactor to read meta once before both blocks. Extract the `const metaPath = join(workstream.path, 'meta.json'); const meta = await readMeta(metaPath);` to before the featureContext block, and wrap it in its own try/catch. Then both featureContext and researchSummaries blocks can reference the same `meta` variable.
